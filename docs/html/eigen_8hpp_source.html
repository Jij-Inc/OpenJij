160

165  T t_list;
166 
167  for(size_t ind=0; ind<graph.get_num_spins(); ind++){
168  for(size_t adj_ind : graph.adj_nodes(ind)){
169  if(ind != adj_ind){
170  t_list.emplace_back(ind, adj_ind, graph.J(ind, adj_ind));
171  }
172  else{
173  t_list.emplace_back(ind, graph.get_num_spins(), graph.h(ind));
174  t_list.emplace_back(graph.get_num_spins(), ind, graph.h(ind));
175  }
176  }

180
