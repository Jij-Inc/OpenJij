49  template<typename RandomNumberEngine>
50  inline static void update(ClIsing& system,
51  RandomNumberEngine& random_number_engine,
52  const utility::ClassicalUpdaterParameter& parameter) {
53  auto urd = std::uniform_real_distribution<>(0, 1.0);
54 
55  // num_spin = system size + additional spin
56  const size_t num_spin = system.spin.size();
57 
58  // 1. update bonds
59  auto union_find_tree = utility::UnionFind(num_spin);
60  for (std::size_t node = 0; node < num_spin; ++node) {
61  for (typename ClIsing::SparseMatrixXx::InnerIterator it(system.interaction, node); it; ++it) {
62  //fetch adjacent node
63  std::size_t adj_node = it.index();
64  //fetch system.interaction(node, adj_node)
65  const FloatType& J = it.value();
66  if (node >= adj_node) continue;
67  //check if bond can be connected
68  if (J * system.spin(node) * system.spin(adj_node) > 0) continue;
69  const auto unite_rate = std::max(static_cast<FloatType>(0.0), static_cast<FloatType>(1.0 - std::exp( - 2.0 * parameter.beta * std::abs(J))));
70  if (urd(random_number_engine) < unite_rate)
71  union_find_tree.unite_sets(node, adj_node);
72  }
73  }
74 
75  // 2. make clusters
76  const auto cluster_map = [num_spin, &union_find_tree](){
77  auto cluster_map = std::unordered_multimap<utility::UnionFind::Node, utility::UnionFind::Node>();
78  for (std::size_t node = 0; node < num_spin; ++node) {
79  cluster_map.insert({union_find_tree.find_set(node), node});
80  }
81  return cluster_map;
82  }();
83 
84  // 3. update spin states in each cluster
85  for (auto&& c : union_find_tree.get_roots()) {
86  const auto range = cluster_map.equal_range(c);
87 
88  // 3.1. decide spin state (flip with the probability 1/2)
89  const FloatType probability = 1.0 / 2.0;
90  if(urd(random_number_engine) < probability){
91  // 3.2. update spin states
92  for (auto itr = range.first, last = range.second; itr != last; ++itr) {
93  const auto idx = itr->second;
94  system.spin(idx) *= -1;
95  }
96  }
97  }
98 
99  return;
100  }
