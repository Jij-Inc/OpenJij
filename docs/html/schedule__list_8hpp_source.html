169  ClassicalScheduleList make_classical_schedule_list(double beta_min, double beta_max, std::size_t one_mc_step, std::size_t num_call_updater) {
170  double r_beta = std::pow(beta_max/beta_min, 1.0/static_cast<double>(num_call_updater - 1));
171  double beta = beta_min;
172 
173  auto schedule_list = ClassicalScheduleList(num_call_updater);
174  for (auto& schedule : schedule_list) {
175  schedule.one_mc_step = one_mc_step;
176  schedule.updater_parameter = ClassicalUpdaterParameter(beta);
177  beta *= r_beta;
178  }
179 
180  return schedule_list;
181  }

192  TransverseFieldScheduleList make_transverse_field_schedule_list(double beta, std::size_t one_mc_step, std::size_t num_call_updater) {
