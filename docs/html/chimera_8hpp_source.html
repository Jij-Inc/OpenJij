153  inline void _checkpair(Index idx1, Index idx2) const{
154  std::int64_t r1,c1,i1,r2,c2,i2;
155  std::tie(r1,c1,i1) = to_rci(idx1);
156  std::tie(r2,c2,i2) = to_rci(idx2);
157  if(
158  !((r1 == r2 && std::abs(c1 - c2) == 1 && i1 == i2 && i1 >= 4) || //horizontal connect
159  (c1 == c2 && std::abs(r1 - r2) == 1 && i1 == i2 && 4 < i1) || //vertical connect
160  (r1 == r2 && c1 == c2 && i1 < 4 && 4 <= i2) || //in-site connect
161  (r1 == r2 && c1 == c2 && i2 < 4 && 4 <= i1) || //in-site connect
162  (r1 == r2 && c1 == c2 && i1 == i2)) //local field
163  ){
164  throw std::runtime_error("invalid index pair "+std::to_string(idx1)+" "+std::to_string(idx2)+" inserted in Chimera");
165  }
166  }

180  Index to_ind(std::int64_t r, std::int64_t c, std::int64_t i) const{
181  //row-major matrix
182 
183  if(!(-1 <= r && r <= static_cast<std::int64_t>(_num_row))){
184  throw std::runtime_error("invalid value r="+std::to_string(r)+" inserted in Chimera");
185  }
186 
187  if(!(-1 <= c && c <= static_cast<std::int64_t>(_num_column))){
188  throw std::runtime_error("invalid value c="+std::to_string(c)+" inserted in Chimera");
189  }
190 
191  if(!(0 <= i && i < static_cast<std::int64_t>(_num_in_chimera))){
192  throw std::runtime_error("invalid value i="+std::to_string(i)+" inserted in Chimera");
193  }
194 
195  return _num_column*_num_in_chimera * mod_r(r) + _num_in_chimera * mod_c(c) + i;
196  }

205  ChimeraIndex to_rci(Index ind) const{
206  if(!(ind < this->get_num_spins())){
207  throw std::runtime_error("invalid value index="+std::to_string(ind)+" inserted in Chimera");
208  }
209 
210  std::int64_t r = ind/(_num_column*_num_in_chimera);
211  ind -= (_num_column*_num_in_chimera)*r;
212  std::int64_t c = ind/_num_in_chimera;
213  ind -= (_num_in_chimera)*c;
214  std::int64_t i = ind;
215  return std::make_tuple(r, c, i);
216  }

225  Chimera(std::size_t num_row, std::size_t num_column, FloatType init_val=0)
226  : Sparse<FloatType>(num_row*num_column*_num_in_chimera, 6+1), _init_val(init_val), _num_row(num_row), _num_column(num_column){
227  //generate sparse graph
228  assert(_num_row >= 1);
229  assert(_num_column >= 1);
230 
231  for(std::size_t r=0; r<_num_row; r++){
232  for(std::size_t c=0; c<_num_column; c++){
233  for(std::size_t i=0; i<_num_in_chimera; i++){
234  //open boundary
235  if(r > 0 && i < 4){
236  //MINUS_R (0<=i<4)
237  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r-1,c,i)) = _init_val;
238  }
239  if(c > 0 && 4 <= i){
240  //MINUS_C (4<=i<8)
241  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c-1,i)) = _init_val;
242  }
243  if(r < num_row-1 && i < 4){
244  //PLUS_R (0<=i<4)
245  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r+1,c,i)) = _init_val;
246  }
247  if(c < num_column-1 && 4 <= i){
248  //PLUS_C (4<=i<8)
249  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c+1,i)) = _init_val;
250  }
251 
252  //inside chimera unit
253 
254  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,(i < 4)?4:0)) = _init_val;
255  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,(i < 4)?5:1)) = _init_val;
256  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,(i < 4)?6:2)) = _init_val;
257  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,(i < 4)?7:3)) = _init_val;
258 
259  //local field
260  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,i)) = _init_val;
261  }
262  }
263  }

274  Chimera(const json& j, std::size_t num_row, std::size_t num_column, FloatType init_val=0) : Chimera<FloatType>(num_row, num_column, init_val){
275  if(!(j["num_variables"] <= num_row*num_column*_num_in_chimera)){
276  throw std::runtime_error("number of system size does not match");
277  }
278  //define bqm with ising variables
279  auto bqm = json_parse<FloatType, cimod::Sparse>(j, false);
280  //interactions
281  for(auto&& elem : bqm.get_quadratic()){
282  const auto& key = elem.first;
283  const auto& val = elem.second;
284  _checkpair(key.first, key.second);
285  this->Sparse<FloatType>::J(key.first, key.second) += val;
286  }

289  const auto& key = elem.first;
