169  void reset_dE() {
170  dE.resize(num_spins);
171  max_dE_ = std::abs(poly_value_list_[0]);//Initialize
172  min_dE_ = std::abs(poly_value_list_[0]);//Initialize
173  if (vartype_ == cimod::Vartype::SPIN) {
174  for (graph::Index i = 0; i < num_spins; ++i) {
175  FloatType temp_energy = 0.0;
176  FloatType temp_abs = 0.0;
177  bool flag = false;
178  for (const auto &it: connected_J_term_index_[i]) {
179  temp_energy += poly_value_list_[it]*spin_sign_poly_[it];
180  temp_abs += std::abs(poly_value_list_[it]);
181  }
182  dE[i] = -2*temp_energy;
183  if (flag && max_dE_ < 2*temp_abs) {
184  max_dE_ = 2*temp_abs;
185  }
186  if (flag && min_dE_ > 2*temp_abs) {
187  min_dE_ = 2*temp_abs;
188  }
189  }
190  }
191  else if (vartype_ == cimod::Vartype::BINARY) {
192  for (graph::Index i = 0; i < num_spins; ++i) {
193  FloatType temp_energy = 0.0;
194  FloatType temp_abs = 0.0;
195  bool flag = false;
196  auto temp_spin = spin[i];
197  for (const auto &it: connected_J_term_index_[i]) {
198  temp_energy += poly_value_list_[it]*Sign(temp_spin)*ZeroOrOne(temp_spin, binary_zero_count_poly_[it]);
199  temp_abs += std::abs(poly_value_list_[it]);
200  flag = true;
201  }
202  dE[i] = temp_energy;
203  if (flag && max_dE_ < temp_abs) {
204  max_dE_ = temp_abs;
205  }
206  if (flag && min_dE_ > temp_abs) {
207  min_dE_ = temp_abs;
208  }
209  }
210  }
211  else {
212  std::stringstream ss;
213  ss << "Unknown vartype detected in " << __func__ << std::endl;
214  throw std::runtime_error(ss.str());
215  }

257  void update_dE_for_binary(std::size_t index) {
258  dE[index] *= -1;
259  const std::size_t begin = crs_row_[index];
260  const std::size_t end = crs_row_[index + 1];
261  for (std::size_t i = begin; i < end; ++i) {

270  assert(init_spins.size() == num_spins);

273  if (vartype_ == cimod::Vartype::SPIN) {
274  for (std::size_t index = 0; index < init_spins.size(); ++index) {
275  if (spin[index] != init_spins[index]) {
276  update_dE_for_spin(index);

293  }

384  void SetUpdateMatrix() {
385  crs_col_.clear();
386  crs_row_.clear();
387  crs_val_.clear();
388  crs_row_.push_back(0);
389  if (vartype_ == cimod::Vartype::SPIN) {
390  crs_sign_p_.clear();
391  for (std::size_t row = 0; row < num_spins; ++row) {
392  std::vector<graph::Index> temp_vec;
393  std::vector<std::vector<FloatType>> temp_vec_val (num_spins);
394  std::vector<std::vector<int8_t*>> temp_vec_sign(num_spins);
395  for (const auto &index: connected_J_term_index_[row]) {
396  auto temp_J_term = poly_value_list_[index];
397  auto temp_sign_p = &spin_sign_poly_[index];
398  for (const auto &col: poly_key_list_[index]) {
399  if (row != col) {
400  temp_vec.push_back(col);
401  temp_vec_val [col].push_back(4*temp_J_term);
402  temp_vec_sign[col].push_back(temp_sign_p);
403  }
404  }
405  }
406  std::sort(temp_vec.begin(), temp_vec.end());
407  temp_vec.erase(std::unique(temp_vec.begin(), temp_vec.end()), temp_vec.end());
408  for (const auto &it: temp_vec) {
409  for (std::size_t i = 0; i < temp_vec_val[it].size(); ++i) {
410  crs_col_.push_back(it);
411  crs_val_.push_back(temp_vec_val[it][i]);
412  crs_sign_p_.push_back(temp_vec_sign[it][i]);
413  }
414  }
415  crs_row_.push_back(crs_col_.size());
416  }
417  }
418  else if (vartype_ == cimod::Vartype::BINARY) {
419  crs_zero_count_p_.clear();
420  for (std::size_t row = 0; row < num_spins; ++row) {
421  std::vector<graph::Index> temp_vec;
422  std::vector<std::vector<FloatType>> temp_vec_val(num_spins);
423  std::vector<std::vector<std::size_t*>> temp_vec_zero_count(num_spins);
424  for (const auto &index: connected_J_term_index_[row]) {
425  auto temp_J_term = poly_value_list_[index];
426  auto temp_zero_count_p = &binary_zero_count_poly_[index];
427  for (const auto &col: poly_key_list_[index]) {
428  if (row != col) {
429  temp_vec.push_back(col);
430  temp_vec_val[col].push_back(temp_J_term);
431  temp_vec_zero_count[col].push_back(temp_zero_count_p);
432  }
433  }
434  }
435  std::sort(temp_vec.begin(), temp_vec.end());
436  temp_vec.erase(std::unique(temp_vec.begin(), temp_vec.end()), temp_vec.end());
437  for (const auto &it: temp_vec) {
438  for (std::size_t i = 0; i < temp_vec_val[it].size(); ++i) {
439  crs_col_.push_back(it);
440  crs_val_.push_back(temp_vec_val[it][i]);
441  crs_zero_count_p_.push_back(temp_vec_zero_count[it][i]);
442  }
443  }
444  crs_row_.push_back(crs_col_.size());
445  }
446  }
447  else {
448  throw std::runtime_error("Unknown vartype detected");
449  }
450  }

453  void SetPolynomialIndex() {
454  connected_J_term_index_.resize(num_spins);
455  if (vartype_ == cimod::Vartype::SPIN) {
456  spin_sign_poly_.resize(num_interactions_);
457  for (std::size_t i = 0; i < num_interactions_; ++i) {
458  graph::Spin spin_multiple = 1;
459  for (const auto &it: poly_key_list_[i]) {
460  spin_multiple *= spin[it];
461  connected_J_term_index_[it].push_back(i);
462  }
463  spin_sign_poly_[i] = static_cast<int8_t>(spin_multiple);
464  }
465  }
466  else if (vartype_ == cimod::Vartype::BINARY) {
467  binary_zero_count_poly_.resize(num_interactions_);
468  for (std::size_t i = 0; i < num_interactions_; ++i) {
469  std::size_t zero_count = 0;
470  graph::Binary binary_multiple = 1;
471  for (const auto &it: poly_key_list_[i]) {
472  binary_multiple *= spin[it];
473  connected_J_term_index_[it].push_back(i);
474  if (spin[it] == 0) {
475  zero_count++;
476  }
477  }
478  binary_zero_count_poly_[i] = zero_count;
479  }
480  }
481  else {
482  throw std::runtime_error("Unknown vartype detected");
483  }

490  void SetPolyKeysAndValues(const cimod::PolynomialKeyList<graph::Index> &input_keys, const cimod::PolynomialValueList<FloatType> &input_values, const bool relabel_flag) {
491  if (input_keys.size() != input_values.size()) {
492  throw std::runtime_error("The sizes of key_list and value_list must match each other");
493  }
494 
495  poly_key_list_ .clear();
496  poly_value_list_.clear();
497 
498  if (relabel_flag) {
499  std::unordered_set<graph::Index> variables;
500  for (std::size_t i = 0; i < input_keys.size(); ++i) {
501  if (input_values[i] != 0.0) {
502  for (const auto &it: input_keys[i]) {
503  variables.emplace(it);
504  }
505  }
506  }
507  std::vector<graph::Index> sorted_variables(variables.begin(), variables.end());
508  std::sort(sorted_variables.begin(), sorted_variables.end());
509 
510  std::unordered_map<graph::Index, graph::Index> relabeld_variables;
511  for (std::size_t i = 0; i < sorted_variables.size(); ++i) {
512  relabeld_variables[sorted_variables[i]] = i;
513  }
514 
515  for (std::size_t i = 0; i < input_keys.size(); ++i) {
516  if (input_values[i] != 0) {
517  std::vector<graph::Index> temp;
518  for (const auto &it: input_keys[i]) {
519  temp.push_back(relabeld_variables[it]);
520  }
521  poly_key_list_.push_back(temp);
522  poly_value_list_.push_back(input_values[i]);
523  }
524  }
525  }
526  else {
527  for (std::size_t i = 0; i < input_keys.size(); ++i) {
528  if (input_values[i] != 0) {
529  poly_key_list_.push_back(input_keys[i]);
530  poly_value_list_.push_back(input_values[i]);
531  }
532  }
533  }
534  num_interactions_ = poly_key_list_.size();
535  }

538  void CheckInitialConditions() const {
539  if (spin.size() != num_spins) {
540  throw std::runtime_error("The number of variables is not equal to the size of the initial spin");
541  }
542  if (poly_key_list_.size() != poly_value_list_.size()) {
543  throw std::runtime_error("The sizes of key_list and value_list must match each other");
544  }
545  if (poly_key_list_.size() <= 0) {
546  throw std::runtime_error("The number of interactions is 0");
547  }
548  if (vartype_ == cimod::Vartype::SPIN) {
549  for (std::size_t i = 0; i < spin.size(); ++i) {
550  if (spin[i] != -1 && spin[i] != 1) {
551  std::stringstream ss;
552  ss << "The variable at " << i << " is " << spin[i] << ".\n";
553  ss << "But the spin variable must be -1 or +1.\n";
554  throw std::runtime_error(ss.str());
555  }
556  }
557  }
558  else if (vartype_ == cimod::Vartype::BINARY) {
559  for (std::size_t i = 0; i < spin.size(); ++i) {
560  if (spin[i] != 0 && spin[i] != 1) {
561  std::stringstream ss;
562  ss << "The variable at " << i << " is " << spin[i] << ".\n";
563  ss << "But the binary variable must be 0 or 1.\n";
564  throw std::runtime_error(ss.str());
565  }
566  }
567  }
568  else {
569  throw std::runtime_error("Unknown vartype detected");
570  }
571  }
