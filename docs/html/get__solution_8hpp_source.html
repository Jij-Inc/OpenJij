57 template<typename GraphType>
58 const graph::Spins get_solution(const system::TransverseIsing<GraphType>& system){
59  std::size_t minimum_trotter = 0;
60  //aliases
61  auto& spins = system.trotter_spins;
62  double energy = 0.0;
63  double min_energy = std::numeric_limits<double>::max();
64  //get number of trotter slices
65  std::size_t num_trotter_slices = system.trotter_spins.cols();
66  for (std::size_t t=0; t<num_trotter_slices; t++){
67  // calculate classical energy in each classical spin
68  energy = spins.col(t).transpose() * system.interaction * spins.col(t);
69  if(energy < min_energy){
70  minimum_trotter = t;
71  min_energy = energy;
72  }
73  }
74 
75  //convert from Eigen::Vector to std::vector
76  graph::Spins ret_spins(system.num_classical_spins);
77  for(std::size_t i=0; i<system.num_classical_spins; i++){
78  ret_spins[i] = spins(i, minimum_trotter);
79  }
80  return ret_spins;
81 }
