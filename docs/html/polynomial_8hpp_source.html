137  FloatType &J(std::vector<Index> &key) {
138  std::sort(key.begin(), key.end());
139  CheckKeyValid(key);
140  if (key.size() > 0) {

243  FloatType calc_energy(const Spins& spins, bool omp_flag = true) const {
244  if(spins.size() != Graph::size()){
245  throw std::out_of_range("Out of range in calc_energy in Polynomial graph.");
246  }
247 
248  FloatType energy = 0.0;
249  std::size_t num_interactions = poly_key_list_.size();
250 
251  if (omp_flag) {
252 #pragma omp parallel for reduction (+: energy)
253  for (std::size_t i = 0; i < num_interactions; ++i) {
254  Spin spin_multiple = 1;
255  for (const auto &index: poly_key_list_[i]) {
256  spin_multiple *= spins[index];
257  if (spin_multiple == 0.0) {
258  break;
259  }
260  }
261  energy += spin_multiple*poly_value_list_[i];
262  }
263  }
264  else {
265  for (std::size_t i = 0; i < num_interactions; ++i) {
266  Spin spin_multiple = 1;
267  for (const auto &index: poly_key_list_[i]) {
268  spin_multiple *= spins[index];
269  if (spin_multiple == 0.0) {
270  break;
271  }
272  }
273  energy += spin_multiple*poly_value_list_[i];
274  }
275  }
276  return energy;
277  }

296  void CheckKeyValid(const std::vector<Index> &key) const {
297  if (0 < key.size()) {
298  //key is assumed to be sorted
299  for (std::size_t i = 0; i < key.size() - 1; ++i) {
300  if (key[i] == key[i + 1]) {
301  throw std::runtime_error("No self-loops allowed");
302  }
303  }
304  }
305  if (key.size() > Graph::size()) {
306  std::stringstream ss;
307  ss << "Too small system size. ";
308  ss << "The degree of the input polynomial interaction is " << key.size();
309  ss << ". But the system size is " << Graph::size() << std::string("\n");
310  throw std::runtime_error(ss.str());
311  }
